Primeira aula: Criação de Function para soma de valores 

            function somar() {
                var tipo1 = document.querySelector('input#t1')
                var tipo2 = document.querySelector('input#t2')
                var resultado = document.querySelector('div#resultado')
                var n1 = Number(tipo1.value)
                var n2 = Number(tipo2.value)
                var s = n1 + n2;
    
                resultado.innerHTML = `A soma entre ${n1} e ${n2} é igual a ${s} `
            }

Segunda aula: Criação de EVentos com o DOM

            var area = document.querySelector("#area")
       
            area.addEventListener("click", clicar)
            area.addEventListener("mouseenter", entrar)
            area.addEventListener("mouseout", sair)

            function clicar() {
                area.innerText = "Clicou";
                area.style.background = "lightblue"
            }

            function entrar() {
                area.innerText = "Entrou"
            }

            function sair() {
                area.innerText = "Saiu"
                area.style.background = "lightpink"
            }
Veja o código funcionando
